Quick Start

# Install from source git clone https://github.com/RedRing2020/mcp-rs.git cd mcp-rs cargo build --release # Run with STDIO transport (for MCP clients) ./target/release/mcp-rs --stdio # Run with TCP transport (for development) ./target/release/mcp-rs --bind-address 127.0.0.1 --port 8080 # Enable debug logging and specific plugins ./target/release/mcp-rs --debug --enable-plugin wordpress --enable-plugin github
